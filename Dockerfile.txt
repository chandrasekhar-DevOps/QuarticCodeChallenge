#From python:3.7.2-alpine3.9
FROM ubuntu:16.04

ENV PYTHONDONTWRITEBYTECODE 1

WORKDIR /usr/src/app/

RUN pip install --no-cache-dir -r requirements.txt

COPY . .

ENTRYPOINT ["/usr/src/app/entrypoint.sh"]
